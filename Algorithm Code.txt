void initMicroTimer(void) {
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;

    TIM2->PSC = (SystemCoreClock / 1000000) - 1; // Prescaler
    TIM2->ARR = 0xFFFFFFFF; // Max count
    TIM2->CR1 |= TIM_CR1_CEN; // Enable timer
}

uint32_t micros(void) {
    return TIM2->CNT;
}

void initStack(Stack* stack) {
    stack->top = -1;
}

int isEmpty(Stack* stack) {
    return stack->top == -1;
}

void push(Stack* stack, double value) {
    if (stack->top < MAX - 1) {
        stack->items[++(stack->top)] = value;
    } else {
        lcd_clear();
        lcd_put_cur(0, 0);
        lcd_send_string("Stack Overflow");
				HAL_Delay(2000);
    }
		return;
}

double pop(Stack* stack) {
    if (isEmpty(stack)) {
        lcd_clear();
        lcd_put_cur(0, 0);
        lcd_send_string("Stack Underflow");
				HAL_Delay(2000);
        return 0;
    }
    return stack->items[(stack->top)--];
}

double peek(Stack* stack) {
    if (isEmpty(stack)) {
        lcd_clear();
        lcd_put_cur(0, 0);
        lcd_send_string("Stack Empty");
			HAL_Delay(2000);
        return 0;
    }
    return stack->items[stack->top];
}

int precedence(char op) {
    if (op == '+' || op == '-') return 1;
    if (op == '*' || op == '/') return 2;
    if (op == '^') return 3;
    return 0;
}

double applyOperator(double a, double b, char op) {
    if (op == '/' && fabs(b) < 1e-12) {
        lcd_clear();
        lcd_put_cur(0, 0);
        lcd_send_string("Divide by Zero");
				HAL_Delay(2000);
        return 0;
    }
    switch (op) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/': return a / b;
        case '^': return powf(a, b);
        default:
            lcd_clear();
            lcd_put_cur(0, 0);
            lcd_send_string("Unknown Op");
						HAL_Delay(2000);
            return 0;
    }
}

double parseNumber(const char* expr, int* i) {
    int sign = 1;
    if (expr[*i] == '-') {
        sign = -1;
        (*i)++;
    }
    double result = 0.0;
    while (isdigit(expr[*i])) {
        result = result * 10 + (expr[*i] - '0');
        (*i)++;
    }
    if (expr[*i] == '.') {
        (*i)++;
        double frac = 0.1;
        while (isdigit(expr[*i])) {
            result += (expr[*i] - '0') * frac;
            frac *= 0.1;
            (*i)++;
        }
    }
    return result * sign;
}

void replaceXWithValue(char* expression, double x_val) {
    char x_str[40];
    snprintf(x_str, sizeof(x_str), "%.12lf", x_val);
    char temp[MAX];
    int k = 0;
    for (int i = 0; expression[i]; i++) {
        if (expression[i] == 'x') {
            for (int j = 0; x_str[j]; j++) {
                if (k < MAX - 1) temp[k++] = x_str[j];
            }
        } else {
            if (k < MAX - 1) temp[k++] = expression[i];
        }
    }
    temp[k] = '\0';
    strcpy(expression, temp);
}

double evaluate(char* expr_input) {
    Stack values, ops;
    initStack(&values);
    initStack(&ops);
    char expr[MAX];
    strcpy(expr, expr_input);
    int i = 0;
    while (expr[i]) {
        if (isspace(expr[i])) {
            i++;
        } else if (expr[i] == '(') {
            push(&ops, expr[i]);
            i++;
        } else if (expr[i] == ')') {
            while (!isEmpty(&ops) && peek(&ops) != '(') {
                double b = pop(&values);
                double a = pop(&values);
                char op = (char)pop(&ops);
                push(&values, applyOperator(a, b, op));
            }
            pop(&ops);  // pop '('
            i++;
        } else if (isdigit(expr[i]) || (expr[i] == '-' && (i == 0 || expr[i - 1] == '(' || strchr("+-*/^", expr[i - 1])))) {
            double val = parseNumber(expr, &i);
            push(&values, val);
        } else {
            while (!isEmpty(&ops) && precedence(peek(&ops)) >= precedence(expr[i])) {
                double b = pop(&values);
                double a = pop(&values);
                char op = (char)pop(&ops);
                push(&values, applyOperator(a, b, op));
            }
            push(&ops, expr[i]);
            i++;
        }
    }
    while (!isEmpty(&ops)) {
        double b = pop(&values);
        double a = pop(&values);
        char op = (char)pop(&ops);
        push(&values, applyOperator(a, b, op));
    }
    return pop(&values);
}

float derivative(char* expr, float x) {
    float h = 1e-5;
    char e1[MAX], e2[MAX];
    strcpy(e1, expr);
    strcpy(e2, expr);
    replaceXWithValue(e1, x + h);
    replaceXWithValue(e2, x - h);
    float f1 = evaluate(e1);
    float f2 = evaluate(e2);
    return (f1 - f2) / (2 * h);
}

float newton(char* expr, float x0) {
    int iter = 0;
    float x = x0;

    while (iter < MAX_ITER) {
        char eval_expr[MAX];
        strcpy(eval_expr, expr);
        replaceXWithValue(eval_expr, x);
        float fx = evaluate(eval_expr);
        float dfx = derivative(expr, x);
			  float x0 = x - fx / dfx;
        if(isnan(dfx)||isinf(dfx)){
            x0=x+1;
        }
        if (fabs(x0 - x) < EPSILON || fabs(dfx) < 1e-10) {
            return x0;
        }

        x = x0;
        iter++;
    }
	return x;

}
